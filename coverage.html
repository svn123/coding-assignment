// This will rotate the characters present in the 2d array if the string is not found in the present run
func rotate_clockwise(puzzle [][]byte) [][]byte {
        var new_puzzle [][]byte
        for i := 0; i < len(puzzle); i++ {
                var new_row []byte
                for j := 0; j < len(puzzle[i]); j++ {
                        new_row = append(new_row, rotation_map[puzzle[i][j]])
                }
                new_puzzle = append(new_puzzle, new_row)
        }
        return new_puzzle
}

// search character in vertical direction
func vertical_search(puzzle [][]byte, word string) (bool, int, int) {
        length := len(word)
        var str string
        for i := 0; i < len(puzzle); i++ {
                for j := 0; j < len(puzzle[i]); j++ {
                        if puzzle[j][i] == word[0] {
                                str = ""
                                for k := 0; k < length; k++ {
                                        if j+k < len(puzzle[i]) {
                                                str += string(puzzle[j+k][i])
                                        }
                                }
                                if str == word {
                                        fmt.Printf("The co-ordinates for first letter of the search word : %v are x : %v , y: %v \n", word, i, j)
                                        return true, i, j
                                }
                        }
                }
        }
        return false, 0, 0
}

// search character in horizontal direction
func horizontal_search(puzzle [][]byte, word string) (bool, int, int) {
        length := len(word)
        var str string
        for i := 0; i < len(puzzle); i++ {
                for j := i; j < len(puzzle[i]); j++ {
                        if puzzle[i][j] == word[0] {
                                str = ""
                                for k := 0; k < length; k++ {
                                        if j+k < len(puzzle[i]) {
                                                str += string(puzzle[i][j+k])
                                        }
                                }

                                if str == word {
                                        fmt.Printf("The co-ordinates for first letter of the search word : %v are  x : %v, y: %v \n", word, i, j)
                                        return true, i, j
                                }
                        }
                }
        }
        return false, 0, 0
}
